class AuditAnswerService {
  final _client = Supabase.instance.client;  // non-late field — throws at construction in tests

  double calculateConformity(
    List<TemplateItem> items,
    Map<String, String> answers,
  ) {
    // Switch over item.responseType with branches:
    //   'ok_nok'    → if (ans == 'ok') earned += item.weight;
    //   'yes_no'    → if (ans == 'yes') earned += item.weight;
    //   'scale_1_5' → earned += (int.tryParse(ans) ?? 0) / 5 * item.weight;
    //   'percentage'→ earned += (double.tryParse(ans) ?? 0) / 100 * item.weight;
    //   'text'      → if (ans.isNotEmpty) earned += item.weight;
    //   'selection' → if (ans.isNotEmpty) earned += item.weight;
    // Empty answers map key or empty string → skipped via `if (ans == null || ans.isEmpty) continue;`
    // Returns 100.0 when totalWeight == 0; otherwise (earned/totalWeight*100).clamp(0.0, 100.0)
  }
}
